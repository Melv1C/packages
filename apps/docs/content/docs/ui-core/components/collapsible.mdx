## Usage

```tsx
import { Collapsible, CollapsibleContent, CollapsibleTrigger } from '@melv1c/ui-core';
```

```tsx
<Collapsible>
  <CollapsibleTrigger>Toggle</CollapsibleTrigger>
  <CollapsibleContent>Content goes here.</CollapsibleContent>
</Collapsible>
```
