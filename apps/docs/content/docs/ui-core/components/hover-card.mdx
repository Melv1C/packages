## Usage

```tsx
import {
  Button,
  HoverCard,
  HoverCardContent,
  HoverCardTrigger,
} from '@melv1c/ui-core';
```

```tsx
<HoverCard>
  <HoverCardTrigger asChild>
    <Button variant="link">Hover Here</Button>
  </HoverCardTrigger>
  <HoverCardContent>Content shown on hover.</HoverCardContent>
</HoverCard>
```
