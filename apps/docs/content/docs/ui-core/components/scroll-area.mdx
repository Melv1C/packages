## Usage

```tsx
import { ScrollArea, Separator } from '@melv1c/ui-core';
```

```tsx
<ScrollArea className="h-72 w-48 rounded-md border">
  <div className="p-4">
    {Array.from({ length: 50 }).map((_, i) => (
      <div key={i}>Item {i + 1}</div>
    ))}
  </div>
</ScrollArea>
```
