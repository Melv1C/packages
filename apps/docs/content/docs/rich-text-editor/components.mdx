## Exported API

```tsx
import {
  BoldButton,
  BulletListButton,
  CodeButton,
  Heading1Button,
  Heading2Button,
  Heading3Button,
  ItalicButton,
  OrderedListButton,
  RedoButton,
  RichTextEditorBuiltIn,
  RichTextEditor,
  RichTextEditorToolbar,
  RichTextEditorContent,
  RichTextEditorHeadingSelect,
  StrikethroughButton,
  UndoButton,
  useRichTextEditor,
} from '@melv1c/rich-text-editor';
```

## Root + toolbar + content

Use these primitives when you need full control.

<ComponentPreview name="rich-text-editor-composable" />

```tsx
import {
  BoldButton,
  ItalicButton,
  RedoButton,
  RichTextEditor,
  RichTextEditorToolbar,
  RichTextEditorContent,
  UndoButton,
} from '@melv1c/rich-text-editor';
import { ButtonGroup } from '@melv1c/ui-core';

<RichTextEditor value={value} onValueChange={setValue}>
  <RichTextEditorToolbar>
    <ButtonGroup>
      <BoldButton />
      <ItalicButton />
    </ButtonGroup>
    <ButtonGroup>
      <UndoButton />
      <RedoButton />
    </ButtonGroup>
  </RichTextEditorToolbar>

  <RichTextEditorContent minHeight={180} maxHeight={320} />
</RichTextEditor>;
```

## Built-in preset

Use `RichTextEditorBuiltIn` for a pre-composed toolbar + content setup:

```tsx
<RichTextEditorBuiltIn value={value} onValueChange={setValue} preset="complete" />
```

## Editor controls

- `BoldButton`, `ItalicButton`, `StrikethroughButton`, `CodeButton` — inline formatting.
- `Heading1Button`, `Heading2Button`, `Heading3Button` — heading toggles.
- `RichTextEditorHeadingSelect` — heading dropdown (with optional paragraph item).
- `BulletListButton`, `OrderedListButton` — list controls.
- `UndoButton`, `RedoButton` — history controls.

Compose controls however you want, for example with `ButtonGroup`:

```tsx
<ButtonGroup>
  <BoldButton />
  <ItalicButton />
  <CodeButton />
</ButtonGroup>

<ButtonGroup>
  <Heading1Button />
  <Heading2Button />
  <Heading3Button />
</ButtonGroup>

<RichTextEditorHeadingSelect levels={[1, 2, 3]} />

<ButtonGroup>
  <BulletListButton />
  <OrderedListButton />
</ButtonGroup>

<ButtonGroup>
  <UndoButton />
  <RedoButton />
</ButtonGroup>
```

## Getting the editor instance

Use `useRichTextEditor()` inside `RichTextEditor` children to build custom actions.

<ComponentPreview name="rich-text-editor-custom-action" />

```tsx title="custom-toolbar-action.tsx"
import { Sparkles } from 'lucide-react';
import { Button } from '@melv1c/ui-core';
import { useRichTextEditor } from '@melv1c/rich-text-editor';

function InsertSignatureButton() {
  const { editor } = useRichTextEditor();

  return (
    <Button
      variant="outline"
      size="icon-sm"
      disabled={!editor}
      onClick={() => editor?.chain().focus().insertContent(' — Team').run()}
      aria-label="Insert signature"
    >
      <Sparkles />
    </Button>
  );
}
```

You can also access the instance from `onValueChange`:

```tsx
<RichTextEditor
  onValueChange={(html, editor) => {
    setValue(html);
    const json = editor.getJSON();
    console.log(json);
  }}
/>
```
